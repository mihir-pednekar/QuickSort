# define C compiler
CC = g++

# define any compile-time flags
CFLAGS = -Wall -c

#remove files
RM = rm -rf

# define the executable file 
MAIN = main

all:    $(MAIN)

$(MAIN): quickSort.o main.o
	$(CC) main.o quickSort.o -o $(MAIN)

main.o: main.cpp
	$(CC) $(CFLAGS) main.cpp

quickSort.o: quickSort.cpp
	$(CC) $(CFLAGS) quickSort.cpp

clean:
	$(RM) *.o $(MAIN)
